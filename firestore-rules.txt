rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users can only read/write their own user document
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Maintenance requests - users can create and read their own, managers can read all
    match /maintenance_requests/{requestId} {
      allow create: if request.auth != null;
      allow read: if request.auth != null && (
        resource.data.userId == request.auth.uid || 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isManager == true
      );
      allow update: if request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isManager == true;
    }
    
    // Notifications - users see their branch notifications, managers see all
    match /notifications/{notificationId} {
      allow read: if request.auth != null && (
        resource.data.branchCode == get(/databases/$(database)/documents/users/$(request.auth.uid)).data.branchCode ||
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isManager == true
      );
      allow create: if request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isManager == true;
      allow update: if request.auth != null && (
        resource.data.branchCode == get(/databases/$(database)/documents/users/$(request.auth.uid)).data.branchCode ||
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isManager == true
      );
    }
  }
}
